import { Clock } from '../src';

import { Canvas, Meta, Story } from '@storybook/addon-docs/blocks';
import { useState } from 'react';
import { format } from 'date-fns';

import { enUS } from 'date-fns/locale';

<Meta title="Clock" component={Clock} />

# Default

<Canvas>
  <Story name="Default">
    {() => {
      const [date, setDate] = useState(null);
      return (
        <div>
          <p>{date ? format(date, 'HH:mm') : 'null'}</p>
          <Clock locale={enUS} date={date} onChange={setDate} />
        </div>
      );
    }}
  </Story>
</Canvas>

# Precision

<Canvas>
  <Story name="Precision">
    {() => {
      const [date, setDate] = useState(null);
      return (
        <div>
          <p>{date ? format(date, 'HH:mm') : 'null'}</p>
          <Clock locale={enUS} date={date} precision={5} onChange={setDate} />
        </div>
      );
    }}
  </Story>
</Canvas>
