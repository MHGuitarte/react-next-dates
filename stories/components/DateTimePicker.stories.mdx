## Children

React next dates don't provide `Input` element.

Like `DatePicker`, `DateTimePicker` children props is a function with parameters to render your own input with your own style.

```tsx
export interface DateTimePickerChildrenProps {
  dateInputProps: DatePickerInputProps;
  timeInputProps: DatePickerInputProps;
  openDatePicker: () => void;
  openTimePicker: () => void;
}

export type DateTimePickerChildren = (props: DateTimePickerChildrenProps) => ReactNode;
```

```tsx
<DateTimePicker locale={enUS}>
  {({ dateInputProps, timeInputProps }) => (
    <>
      <input {...dateInputProps} />
      <input {...timeInputProps} />
    </>
  )}
</DateTimePicker>
```
